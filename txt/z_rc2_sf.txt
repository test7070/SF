set @t_noshowina = case when '#non'=[24] then '' else [24] end  
 
declare @tmp table(
	gno nvarchar(1),
	rr int,
	recno int,
	page int,
	datea nvarchar(10),
	noa nvarchar(50),
	tggno nvarchar(50),
	comp nvarchar(100),
	ordeno nvarchar(100),
	tax float,
	tranmoney float,
	noq nvarchar(50),
	productno nvarchar(50),
	product nvarchar(max),
	mount float,
	price float,
	total float,
	tmount float,
	ttotal float,
	money float,
	ghref nvarchar(max),
	memo nvarchar(max)
)
--明細
if(@t_noshowina='1')
begin
	insert @tmp
	select '2','','','',a.datea,a.noa,a.tggno,a.tgg
	,SUBSTRING(a.transtart,0,CHARINDEX('@',a.transtart)),tax,tranmoney
	,ROW_NUMBER()over(partition by a.noa order by a.noa,b.noq),productno
	,case when len(b.ucolor)=0 then '' else CONVERT(nvarchar,b.ucolor) +replicate('　',(4-len(b.ucolor)))+' '+case when isnull(b.spec,'')!='' then b.spec+' ' else '' end end
	+b.product+b.size+case when isnull(b.lengthb,0)!=0 then ' '+cast(b.lengthb as nvarchar(20))+'M' else '' end
	,case when UPPER(isnull(d.unit,''))='KG' then b.weight else b.mount end
	,b.price,b.total,case when UPPER(isnull(d.unit,''))='KG' then b.weight else 0 end
	,b.total,a.total,'rc2_sf?noa=$noa?'+a.accy,a.memo
	from view_rc2 a left join view_rc2s b on a.noa=b.noa 
	left join ucc d on b.product=d.product
	where (a.datea between @t_bdate and @t_edate) and 
	(a.tggno between @t_btggno and @t_etggno)  and
	(isnull(b.storeno,'') between @t_bstoreno and @t_estoreno) and
	(len(@rc2typea)=0 or @rc2typea=a.typea) and
	(len(@t_product)=0 or @t_product=b.product)and
	(len(@t_spec)=0 or @t_spec=b.spec) and
	(len(@t_size)=0 or @t_size=b.size) and
	(len(@t_class)=0 or @t_class=b.class) and
	(isnull(b.lengthb,0) between @t_blengthb and @t_elengthb) and 
	(len(@t_qno)=0 or dbo.split(dbo.split(a.transtart,'##',0),'@',0) = @t_qno or dbo.split(dbo.split(a.transtart,'##',1),'@',0) = @t_qno)
	and not exists (select noa from view_ina where noa=a.noa)
end
else
begin		
	insert @tmp
	select '2','','','',a.datea,a.noa,a.tggno,a.tgg
	,SUBSTRING(a.transtart,0,CHARINDEX('@',a.transtart)),tax,tranmoney
	,ROW_NUMBER()over(partition by a.noa order by a.noa,b.noq),productno
	,case when len(b.ucolor)=0 then '' else CONVERT(nvarchar,b.ucolor) +replicate('　',(4-len(b.ucolor)))+' '+case when isnull(b.spec,'')!='' then b.spec+' ' else '' end end
	+b.product+b.size+case when isnull(b.lengthb,0)!=0 then ' '+cast(b.lengthb as nvarchar(20))+'M' else '' end
	,case when UPPER(isnull(d.unit,''))='KG' then b.weight else b.mount end
	,b.price,b.total,case when UPPER(isnull(d.unit,''))='KG' then b.weight else 0 end
	,b.total,a.total
	,'rc2_sf?noa=$noa?'+a.accy,a.memo
	from view_rc2 a left join view_rc2s b on a.noa=b.noa 
	left join ucc d on b.product=d.product
	where (a.datea between @t_bdate and @t_edate) and 
	(a.tggno between @t_btggno and @t_etggno)  and
	(isnull(b.storeno,'') between @t_bstoreno and @t_estoreno) and
	(len(@rc2typea)=0 or @rc2typea=a.typea) and
	(len(@t_product)=0 or @t_product=b.product)and
	(len(@t_spec)=0 or @t_spec=b.spec) and
	(len(@t_size)=0 or @t_size=b.size) and
	(len(@t_class)=0 or @t_class=b.class) and
	(isnull(b.lengthb,0) between @t_blengthb and @t_elengthb) and 
	(len(@t_qno)=0 or dbo.split(dbo.split(a.transtart,'##',0),'@',0) = @t_qno or dbo.split(dbo.split(a.transtart,'##',1),'@',0) = @t_qno)
end

--小計
insert @tmp
select '1' ,0,0,0,MAX(datea),noa,tggno,MAX(comp),ordeno
,MAX(tax),MAX(tranmoney),'','','',null,null,null
,SUM(tmount)
,SUM(total),max(money),MAX(ghref),''
from @tmp where datea between @t_bdate and @t_edate and gno ='2'
group by noa,tggno,ordeno

--請款金額
update @tmp
set money=money+tranmoney+tax
where gno='1'

--項次
update a
set rr=rx
from (select ROW_NUMBER()over(partition by gno,tggno order by noa)rx,rr from @tmp where gno='1')a

--合計
insert @tmp(gno,tggno,ordeno,noa,tranmoney,tmount,ttotal,money)
select '3',tggno,ordeno,CHAR(255),SUM(tranmoney),SUM(tmount),sum(ttotal),SUM(tranmoney)+SUM(tax)+SUM(ttotal)
from @tmp where gno='1'
group by tggno,ordeno

--重新排序
update a
set recno=recno2
from (select recno,ROW_NUMBER()over (partition by tggno,ordeno order by noa,gno,rr)recno2 from @tmp)a

--插入換頁
declare @pageline float=37
declare @page int=ceiling(cast((select count(*) from @tmp) as float)/@pageline)
declare @tpage int=0

update a
set page=ceiling(cast(recno as float)/@pageline)
from (select page,recno from @tmp)a

--分頁
insert @tmp(gno,tggno,ordeno,page,noa)
select '4',tggno,ordeno,page,CHAR(255)
from @tmp 
group by tggno,ordeno,page

select 
dbo.getComma(total,0) total
,dbo.getComma(ttotal,0) ttotal
,dbo.getComma(price,3) price
,dbo.getComma(mount,0) mount
,dbo.getComma(tmount,0) tmount
,dbo.getComma(tranmoney,0) tranmoney
,dbo.getComma(money,0) money
,* from @tmp order by tggno,ordeno,page,noa,gno
;
